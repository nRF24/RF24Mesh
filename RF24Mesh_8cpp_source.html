292 bool RF24Mesh::requestAddress(uint8_t level)
293 {
294  RF24NetworkHeader header(MESH_MULTICAST_ADDRESS, NETWORK_POLL);
295  //Find another radio, starting with level 0 multicast
296  IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Poll\n"), millis()));
297  network.multicast(header, 0, 0, level);
298 
299  uint32_t timr = millis();
300  #define MESH_MAXPOLLS 4
301  uint16_t contactNode[MESH_MAXPOLLS];
302  uint8_t pollCount = 0;
303 
304  while (1) {
305  #if defined (MESH_DEBUG)
306  bool goodSignal = radio.testRPD();
307  #endif
308 
309  if (network.update() == NETWORK_POLL) {
310 
311  memcpy(&contactNode[pollCount], &network.frame_buffer[0], sizeof(uint16_t));
312  if (pollCount > 0 && contactNode[pollCount] != contactNode[pollCount - 1]) { //Drop duplicate polls to help prevent duplicate requests
313  ++pollCount;
314  }
315  else{
316  ++pollCount;
317  }
318 
319  IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Poll %c -64dbm\n"), millis(), (goodSignal ? '>' : '<')));
320  } // end if
321 
322  if (millis() - timr > 55 || pollCount >= MESH_MAXPOLLS ) {
323  if (!pollCount) {
324  IF_MESH_DEBUG(printf_P(PSTR("%u: MSH No poll from level %d\n"), millis(), level));
325  return 0;
326  }
327  else {
328  IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Poll OK\n"), millis()));
329  break;
330  }
331  }
332  MESH_CALLBACK
333  } // end while
334 
335 
336  IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Got poll from level %d count %d\n"), millis(), level, pollCount));
337 
338  bool gotResponse = 0;
339  for (uint8_t i = 0; i < pollCount; i++) {
340  // Request an address via the contact node
341  header.type = NETWORK_REQ_ADDRESS;
342  header.reserved = _nodeID;
343  header.to_node = contactNode[i];
344 
345  // Do a direct write (no ack) to the contact node. Include the nodeId and address.
346  network.write(header, 0, 0, contactNode[i]);
347 
348  IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Request address from: 0%o\n"), millis(), contactNode[i]));
349 
350  timr = millis();
351 
352  while (millis() - timr < 225) {
353  if (network.update() == NETWORK_ADDR_RESPONSE) {
354  if (network.frame_buffer[7] == _nodeID ) {
355  uint16_t newAddy = 0;
356  memcpy(&newAddy, &network.frame_buffer[sizeof(RF24NetworkHeader)], sizeof(newAddy));
357  uint16_t mask = 0xFFFF;
358  newAddy &= ~(mask << (3 * getLevel(contactNode[i]))); // Get the level of contact node. Multiply by 3 to get the number of bits to shift (3 per digit)
359  if (newAddy == contactNode[i]) { // Then shift the mask by this much, and invert it bitwise. Apply the mask to the newly received
360  i = pollCount; // address to evalute whether 'subnet' of the assigned address matches the contact node address.
361  gotResponse = 1;
362  break;
363  }
364  }
365  }
366  MESH_CALLBACK
367  }
368  } // end for
369 
370  if (!gotResponse) {
371  return 0;
372  }
373 
374  uint16_t newAddress=0;
375  memcpy(&newAddress, network.frame_buffer + sizeof(RF24NetworkHeader), sizeof(newAddress));
376 
377  IF_MESH_DEBUG(printf_P(PSTR("Set address 0%o rcvd 0%o\n"), mesh_address, newAddress));
378  mesh_address = newAddress;
379 
380  radio.stopListening();
381  network.begin(mesh_address);
382 
383  // getNodeID() doesn't use auto-ack; do a double-check to manually retry 1 more time
384  if (getNodeID(mesh_address) != _nodeID) {
385  if (getNodeID(mesh_address) != _nodeID) {
386  beginDefault();
387  return 0;
388  }
389  }
390  return 1;
391 }

482 void RF24Mesh::DHCP()
483 {
484  if (doDHCP)
485  doDHCP = false;
486  else
487  return;
488 
489  RF24NetworkHeader header;
490  memcpy(&header, network.frame_buffer, sizeof(RF24NetworkHeader));
491 
492  uint16_t newAddress;
493 
494  // Get the unique id of the requester (ID is in header.reserved)
495  if (!header.reserved || header.type != NETWORK_REQ_ADDRESS) {
496  IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Invalid id or type rcvd\n"), millis()));
497  return;
498  }
499 
500  uint16_t fwd_by = 0;
501  uint8_t shiftVal = 0;
502  bool extraChild = false;
503 
504  if (header.from_node != MESH_DEFAULT_ADDRESS) {
505  fwd_by = header.from_node;
506  uint16_t m = fwd_by;
507  uint8_t count = 0;
508 
509  while(m) { //Octal addresses convert nicely to binary in threes. Address 03 = B011 Address 033 = B011011
510  m >>= 3; //Find out how many digits are in the octal address
511  count += 3;
512  }
513  shiftVal = count; //Now we know how many bits to shift when adding a child node 1-5 (B001 to B101) to any address
514  }
515  else {
516  //If request is coming from level 1, add an extra child to the master
517  extraChild = 1;
518  }
519 
520  // IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Rcv addr req from_id %d\n"), millis(), header.reserved));
521 
522  for (int i = MESH_MAX_CHILDREN + extraChild; i > 0; i--) { // For each of the possible addresses (5 max)
523 
524  bool found = false;
525  newAddress = fwd_by | (i << shiftVal);
526 
527  if (newAddress == MESH_DEFAULT_ADDRESS) { continue; }
528 
529  for (uint8_t i = 0; i < addrListTop; i++) {
530  IF_MESH_DEBUG_MINIMAL(printf_P(PSTR("ID: %d ADDR: 0%o\n"), addrList[i].nodeID, addrList[i].address));
531  if (addrList[i].address == newAddress && addrList[i].nodeID != header.reserved) {
532  found = true;
533  break;
534  }
535  } // 3 conditions: 1. address in list = assigned to somebody else (bad); 2. address in list = assigned to this nodeID (ok); 3. address not in list (ok)
536 
537  if (!found) {
538  header.type = NETWORK_ADDR_RESPONSE;
539  header.to_node = header.from_node;
540  //This is a routed request to 00
541 
542  setAddress(header.reserved, newAddress);
543  // without this delay, address renewal fails for children with slower execution speed
544  #if defined (SLOW_ADDR_POLL_RESPONSE)
545  delay(SLOW_ADDR_POLL_RESPONSE);
546  #endif // defined (SLOW_ADDR_POLL_RESPONSE)
547  if (header.from_node != MESH_DEFAULT_ADDRESS) { //Is NOT node 01 to 05
548  //delay(2);
549  if (!network.write(header, &newAddress, sizeof(newAddress))) {
550  network.write(header, &newAddress, sizeof(newAddress));
551  }
552  }
553  else {
554  //delay(2);
555  network.write(header, &newAddress, sizeof(newAddress), header.to_node);
556  }
557 
558  IF_MESH_DEBUG(printf_P(PSTR("Sent to 0%o phys: 0%o new: 0%o id: %d\n"), header.to_node, MESH_DEFAULT_ADDRESS, newAddress, header.reserved));
559  break;
560  }
561  else {
562  IF_MESH_DEBUG(printf_P(PSTR("not allocated\n")));
563  }
564  } // end for
565 }
