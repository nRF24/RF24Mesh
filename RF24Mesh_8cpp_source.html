60uint8_t RF24Mesh::update()
61{
62 uint8_t type = network.update();
63 if (mesh_address == MESH_DEFAULT_ADDRESS) return type;
64
65#if !defined(MESH_NOMASTER)
66 if (type == NETWORK_REQ_ADDRESS) {
67 doDHCP = 1;
68 }
69
70 if (!getNodeID()) {
71 if ((type == MESH_ADDR_LOOKUP || type == MESH_ID_LOOKUP)) {
72 RF24NetworkHeader* header = (RF24NetworkHeader*)(&network.frame_buffer);
73 header->to_node = header->from_node;
74
75 int16_t returnAddr = 0;
76 if (type == MESH_ADDR_LOOKUP) {
77 returnAddr = getAddress(network.frame_buffer[sizeof(RF24NetworkHeader)]);
78 network.write(*header, &returnAddr, sizeof(returnAddr));
79 }
80 else {
81 int16_t addr = 0;
82 memcpy(&addr, &network.frame_buffer[sizeof(RF24NetworkHeader)], sizeof(addr));
83 returnAddr = getNodeID(addr);
84 network.write(*header, &returnAddr, sizeof(returnAddr));
85 }
86 }
87 else if (type == MESH_ADDR_RELEASE) {
88 uint16_t* fromAddr = (uint16_t*)network.frame_buffer;
89 for (uint8_t i = 0; i < addrListTop; i++) {
90 if (addrList[i].address == *fromAddr) {
91 addrList[i].address = 0;
92 }
93 }
94 }
95 }
96#endif
97
98 return type;

269uint16_t RF24Mesh::renewAddress(uint32_t timeout)
270{
271 if (radio.available()) network.update();
272
273 uint8_t reqCounter = 0;
274 uint8_t totalReqs = 0;
275
276 beginDefault();
277
278 uint32_t start = millis();
279 while (!requestAddress(reqCounter)) {
280 if (millis() - start > timeout) break;
281
282 uint32_t timeoutInternal = 50 + ((totalReqs + 1) * (reqCounter + 1)) * 2;
283 uint32_t startInternal = millis();
284 while (millis() - startInternal < timeoutInternal) {
285 MESH_CALLBACK
286 delay(1);
287 }
288 reqCounter++;
289 reqCounter = reqCounter % 4;
290 totalReqs++;
291 totalReqs = totalReqs % 10;
292 }
293 return mesh_address;
294}

298bool RF24Mesh::requestAddress(uint8_t level)
299{
300 RF24NetworkHeader header(MESH_MULTICAST_ADDRESS, NETWORK_POLL);
301 //Find another radio, starting with level 0 multicast
302 IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Poll\n"), millis()));
303 network.multicast(header, 0, 0, level);
304
305 uint32_t timr = millis();
306#define MESH_MAXPOLLS 4
307 uint16_t contactNode[MESH_MAXPOLLS];
308 uint8_t pollCount = 0;
309
310 while (1) {
311#if defined(MESH_DEBUG)
312 bool goodSignal = radio.testRPD();
313#endif
314
315 if (network.update() == NETWORK_POLL) {
316
317 memcpy(&contactNode[pollCount], &network.frame_buffer[0], sizeof(uint16_t));
318 if (pollCount > 0 && contactNode[pollCount] != contactNode[pollCount - 1]) { //Drop duplicate polls to help prevent duplicate requests
319 ++pollCount;
320 }
321 else {
322 ++pollCount;
323 }
324
325 IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Poll %c -64dbm\n"), millis(), (goodSignal ? '>' : '<')));
326 } // end if
327
328 if (millis() - timr > 55 || pollCount >= MESH_MAXPOLLS) {
329 if (!pollCount) {
330 IF_MESH_DEBUG(printf_P(PSTR("%u: MSH No poll from level %d\n"), millis(), level));
331 return 0;
332 }
333 else {
334 IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Poll OK\n"), millis()));
335 break;
336 }
337 }
338 MESH_CALLBACK
339 } // end while
340
341 IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Got poll from level %d count %d\n"), millis(), level, pollCount));
342
343 bool gotResponse = 0;
344 for (uint8_t i = 0; i < pollCount; i++) {
345 // Request an address via the contact node
346 header.type = NETWORK_REQ_ADDRESS;
347 header.reserved = _nodeID;
348 header.to_node = contactNode[i];
349
350 // Do a direct write (no ack) to the contact node. Include the nodeId and address.
351 network.write(header, 0, 0, contactNode[i]);
352
353 IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Request address from: 0%o\n"), millis(), contactNode[i]));
354
355 timr = millis();
356
357 while (millis() - timr < 225) {
358 if (network.update() == NETWORK_ADDR_RESPONSE) {
359 if (network.frame_buffer[7] == _nodeID) {
360 uint16_t newAddy = 0;
361 memcpy(&newAddy, &network.frame_buffer[sizeof(RF24NetworkHeader)], sizeof(newAddy));
362 uint16_t mask = 0xFFFF;
363 newAddy &= ~(mask << (3 * getLevel(contactNode[i]))); // Get the level of contact node. Multiply by 3 to get the number of bits to shift (3 per digit)
364 if (newAddy == contactNode[i]) { // Then shift the mask by this much, and invert it bitwise. Apply the mask to the newly received
365 i = pollCount; // address to evalute whether 'subnet' of the assigned address matches the contact node address.
366 gotResponse = 1;
367 break;
368 }
369 }
370 }
371 MESH_CALLBACK
372 }
373 } // end for
374
375 if (!gotResponse) {
376 return 0;
377 }
378
379 uint16_t newAddress = 0;
380 memcpy(&newAddress, network.frame_buffer + sizeof(RF24NetworkHeader), sizeof(newAddress));
381
382 IF_MESH_DEBUG(printf_P(PSTR("Set address 0%o rcvd 0%o\n"), mesh_address, newAddress));
383 mesh_address = newAddress;
384
385 radio.stopListening();
386 network.begin(mesh_address);
387
388 // getNodeID() doesn't use auto-ack; do a double-check to manually retry 1 more time
389 if (getNodeID(mesh_address) != _nodeID) {
390 if (getNodeID(mesh_address) != _nodeID) {
391 beginDefault();
392 return 0;
393 }
394 }
395 return 1;
396}

415void RF24Mesh::setAddress(uint8_t nodeID, uint16_t address, bool searchBy)
416{
417 //Look for the node in the list
418 for (uint8_t i = 0; i < addrListTop; i++) {
419 if (searchBy == false) {
420 if (addrList[i].nodeID == nodeID) {
421 addrList[i].address = address;
422 #if defined(__linux) && !defined(__ARDUINO_X86__)
423 saveDHCP();
424 #endif
425 return; //Found & set, complete
426 }
427 }
428 else { // Search by address, set the nodeID
429 if (addrList[i].address == address) {
430 //printf("*** Addr 0%o Found, reassign fr ID %d to ID %d ***\n", addrList[i].address, addrList[i].nodeID, nodeID);
431 addrList[i].nodeID = nodeID;
432 #if defined(__linux) && !defined(__ARDUINO_X86__)
433 saveDHCP();
434 #endif
435 return;
436 }
437 }
438 }
439
440 if (addrListTop > 0 && addrListTop % MESH_MEM_ALLOC_SIZE == 0) {
441 addrList = (addrListStruct*)realloc(addrList, (addrListTop + MESH_MEM_ALLOC_SIZE) * sizeof(addrListStruct));
442 }
443 addrList[addrListTop].address = address;
444 addrList[addrListTop++].nodeID = nodeID; //Set the value AND increment Top without another line of code
445 #if defined(__linux) && !defined(__ARDUINO_X86__)
446 saveDHCP();
447 #endif
448}

488
489void RF24Mesh::DHCP()
490{
491 if (doDHCP)
492 doDHCP = false;
493 else
494 return;
495
496 RF24NetworkHeader header;
497 memcpy(&header, network.frame_buffer, sizeof(RF24NetworkHeader));
498
499 uint16_t newAddress;
500
501 // Get the unique id of the requester (ID is in header.reserved)
502 if (!header.reserved || header.type != NETWORK_REQ_ADDRESS) {
503 IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Invalid id or type rcvd\n"), millis()));
504 return;
505 }
506
507 uint16_t fwd_by = 0;
508 uint8_t shiftVal = 0;
509 bool extraChild = false;
510
511 if (header.from_node != MESH_DEFAULT_ADDRESS) {
512 fwd_by = header.from_node;
513 uint16_t m = fwd_by;
514 uint8_t count = 0;
515
516 while (m) { //Octal addresses convert nicely to binary in threes. Address 03 = B011 Address 033 = B011011
517 m >>= 3; //Find out how many digits are in the octal address
518 count += 3;
519 }
520 shiftVal = count; //Now we know how many bits to shift when adding a child node 1-5 (B001 to B101) to any address
521 }
522 else {
523 //If request is coming from level 1, add an extra child to the master
524 extraChild = 1;
525 }
526
527 // IF_MESH_DEBUG(printf_P(PSTR("%u: MSH Rcv addr req from_id %d\n"), millis(), header.reserved));
528
529 for (int i = MESH_MAX_CHILDREN + extraChild; i > 0; i--) { // For each of the possible addresses (5 max)
530
531 bool found = false;
532 newAddress = fwd_by | (i << shiftVal);
533
534 if (newAddress == MESH_DEFAULT_ADDRESS) continue;
535
536 for (uint8_t i = 0; i < addrListTop; i++) {
537 IF_MESH_DEBUG_MINIMAL(printf_P(PSTR("ID: %d ADDR: 0%o\n"), addrList[i].nodeID, addrList[i].address));
538 if (addrList[i].address == newAddress && addrList[i].nodeID != header.reserved) {
539 found = true;
540 break;
541 }
542 } // 3 conditions: 1. address in list = assigned to somebody else (bad); 2. address in list = assigned to this nodeID (ok); 3. address not in list (ok)
543
544 if (!found) {
545 header.type = NETWORK_ADDR_RESPONSE;
546 header.to_node = header.from_node;
547 //This is a routed request to 00
548
549 setAddress(header.reserved, newAddress);
550 // without this delay, address renewal fails for children with slower execution speed
551 #if defined(SLOW_ADDR_POLL_RESPONSE)
552 delay(SLOW_ADDR_POLL_RESPONSE);
553 #endif // defined (SLOW_ADDR_POLL_RESPONSE)
554
555 if (header.from_node != MESH_DEFAULT_ADDRESS) { //Is NOT node 01 to 05
556 //delay(2);
557 if (!network.write(header, &newAddress, sizeof(newAddress))) {
558 network.write(header, &newAddress, sizeof(newAddress));
559 }
560 }
561 else {
562 //delay(2);
563 network.write(header, &newAddress, sizeof(newAddress), header.to_node);
564 }
565
566 IF_MESH_DEBUG(printf_P(PSTR("Sent to 0%o phys: 0%o new: 0%o id: %d\n"), header.to_node, MESH_DEFAULT_ADDRESS, newAddress, header.reserved));
567 break;
568 }
569 else {
570 IF_MESH_DEBUG(printf_P(PSTR("not allocated\n")));
571 }
572 } // end for
573}
